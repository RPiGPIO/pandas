# Shows only first 2 documents
 db.students.find().limit(2)


# Skips first document and shows remaining
 db.students.find().skip(1)


# Shows only name and city, hides _id
 db.students.find({}, {name:1, city:1, _id:0})

# Creates index on name field for faster search
 db.students.createIndex({name:1})

# Shows all indexes
 db.students.getIndexes()

# Deletes complete students collection
 db.students.drop()


# Deletes current database completely
 db.dropDatabase()

 ])

 db.studentdata.find().pretty()

 db.studentdata.find().sort({marks:-1})
